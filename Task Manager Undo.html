<script>

let tasks = [

  { id:1, text:'Design the landing page hero', priority:'high', done:false },

  { id:2, text:'Write unit tests for auth module', priority:'med', done:false },

  { id:3, text:'Update project README', priority:'low', done:true },

  { id:4, text:'Code review — PR #42', priority:'high', done:false },

];

let nextId = 5;

let currentFilter = 'all';

let editingId = null;


const undoStack = [];

const redoStack = [];

let undoCount = 0;

let redoCount = 0;


function snap() { return JSON.parse(JSON.stringify(tasks)); }


function pushUndo(prev) {

  undoStack.push(prev);

  redoStack.length = 0;  // clear redo on new action

}


function undo() {

  if (!undoStack.length) return;

  redoStack.push(snap());

  tasks = undoStack.pop();

  undoCount++;

  logAction('undo', 'Undo performed');

  renderAll();

  showToast('↺ Undone');

}


function redo() {

  if (!redoStack.length) return;

  undoStack.push(snap());

  tasks = redoStack.pop();

  redoCount++;

  logAction('redo', 'Redo performed');

  renderAll();

  showToast('↻ Redone');

}


function addTask() {

  const inp = document.getElementById('taskInput');

  const pri = document.getElementById('prioritySelect').value;

  const text = inp.value.trim();

  if (!text) { inp.focus(); return; }

  const prev = snap();

  tasks.push({ id: nextId++, text, priority: pri, done: false });

  pushUndo(prev);

  logAction('add', `Added "${shorten(text)}"`);

  inp.value = '';

  renderAll();

}


function deleteTask(id) {

  const t = tasks.find(t => t.id === id);

  const prev = snap();

  tasks = tasks.filter(t => t.id !== id);

  pushUndo(prev);

  logAction('delete', `Deleted "${shorten(t?.text || '')}"`);

  renderAll();

  showToast('Task removed');

}


function toggleDone(id) {

  const prev = snap();

  const t = tasks.find(t => t.id === id);

  if (t) t.done = !t.done;

  pushUndo(prev);

  logAction('toggle', `${t?.done ? 'Checked' : 'Unchecked'} "${shorten(t?.text || '')}"`);

  renderAll();

}


function startEdit(id) {

  editingId = id;

  renderAll();

  const inp = document.querySelector('.edit-input');

  if (inp) { inp.focus(); inp.setSelectionRange(inp.value.length, inp.value.length); }

}


function commitEdit(id, val) {

  if (editingId !== id) return;

  editingId = null;

  const text = val.trim();

  if (!text) { renderAll(); return; }

  const t = tasks.find(t => t.id === id);

  if (t && t.text !== text) {

    const prev = snap();

    t.text = text;

    pushUndo(prev);

    logAction('edit', `Edited → "${shorten(text)}"`);

  }

  renderAll();

}


function setFilter(f) {

  currentFilter = f;

  document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));

  document.querySelectorAll('.filter-btn').forEach(el => el.classList.remove('active'));

  const navMap = { all:0, active:1, done:2, high:3, med:4, low:5 };

  const navItems = document.querySelectorAll('.nav-item');

  if (navMap[f] !== undefined) navItems[navMap[f]]?.classList.add('active');

  const fb = document.getElementById('fb-' + f);

  if (fb) fb.classList.add('active');

  document.getElementById('filterLabel').textContent = '— ' + f;

  renderAll();

}


function shorten(s) { return s.length > 22 ? s.slice(0, 21) + '…' : s; }

function esc(s) { return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }


function filteredTasks() {

  switch (currentFilter) {

    case 'active': return tasks.filter(t => !t.done);

    case 'done':   return tasks.filter(t => t.done);

    case 'high':   return tasks.filter(t => t.priority === 'high');

    case 'med':    return tasks.filter(t => t.priority === 'med');

    case 'low':    return tasks.filter(t => t.priority === 'low');

    default:       return tasks;

  }

}


function renderAll() {

  renderTasks();

  renderStacks();

  renderStats();

  document.getElementById('undoBtn').disabled = undoStack.length === 0;

  document.getElementById('redoBtn').disabled = redoStack.length === 0;

}


function renderTasks() {

  const list = document.getElementById('taskList');

  const ft = filteredTasks();

  if (!ft.length) {

    list.innerHTML = `<div class="empty-state"><div class="emoji">✦</div><p>No tasks here yet</p></div>`;

    return;

  }

  list.innerHTML = ft.map(t => {

    const cbClass = t.done ? 'task-cb checked' : 'task-cb';

    const checkSvg = `<svg class="check-icon" width="9" height="9" viewBox="0 0 10 10" fill="none" stroke="#0f0f11" stroke-width="2"><path d="M2 5l2.5 2.5 4-4"/></svg>`;

    const badge = `<span class="priority-badge p-${t.priority}">${t.priority}</span>`;

    const textPart = editingId === t.id

      ? `<input class="edit-input" value="${esc(t.text)}"

           onblur="commitEdit(${t.id},this.value)"

           onkeydown="if(event.key==='Enter')this.blur();if(event.key==='Escape'){editingId=null;renderAll()}">`

      : `<span class="task-text${t.done?' done':''}" ondblclick="startEdit(${t.id})" title="Double-click to edit">${esc(t.text)}</span>`;

    return `<div class="task-card${t.done?' done-card':''}">

      <div class="${cbClass}" onclick="toggleDone(${t.id})">${checkSvg}</div>

      ${textPart}

      ${badge}

      <div class="task-actions">

        <button class="icon-btn" onclick="startEdit(${t.id})" title="Edit">

          <svg width="13" height="13" viewBox="0 0 16 16" fill="none" stroke="currentColor" stroke-width="1.5"><path d="M11.5 2.5a1.5 1.5 0 012.1 2.1L5 13.2l-3 .8.8-3L11.5 2.5z"/></svg>

        </button>

        <button class="icon-btn del" onclick="deleteTask(${t.id})" title="Delete">

          <svg width="13" height="13" viewBox="0 0 16 16" fill="none" stroke="currentColor" stroke-width="1.5"><path d="M4 4l8 8M12 4l-8 8"/></svg>

        </button>

      </div>

    </div>`;

  }).join('');

}


function actionLabel(curr, prev) {

  if (curr.length > prev.length) return 'add';

  if (curr.length < prev.length) return 'delete';

  for (let i = 0; i < Math.min(curr.length, prev.length); i++) {

    if (curr[i].done !== prev[i].done) return `toggle "${shorten(curr[i].text)}"`;

    if (curr[i].text !== prev[i].text) return `edit "${shorten(curr[i].text)}"`;

  }

  return 'change';

}


function renderStacks() {

  const uEl = document.getElementById('undoStackEl');

  const rEl = document.getElementById('redoStackEl');

  if (!undoStack.length) {

    uEl.innerHTML = '<div class="stack-empty">— empty —</div>';

  } else {

    uEl.innerHTML = [...undoStack].reverse().map((prev, idx) => {

      const isTop = idx === 0;

      const label = actionLabel(tasks, prev);

      return `<div class="stack-entry ${isTop?'top-undo':''}">

        <div class="stack-dot"></div>

        <span class="stack-entry-label">${label}</span>

        ${isTop?'<span class="stack-top-tag">TOP</span>':''}

      </div>`;

    }).join('');

  }

  if (!redoStack.length) {

    rEl.innerHTML = '<div class="stack-empty">— empty —</div>';

  } else {

    rEl.innerHTML = [...redoStack].reverse().map((_, idx) => {

      const isTop = idx === 0;

      return `<div class="stack-entry ${isTop?'top-redo':''}">

        <div class="stack-dot"></div>

        <span class="stack-entry-label">snapshot ${redoStack.length - idx}</span>

        ${isTop?'<span class="stack-top-tag">TOP</span>':''}

      </div>`;

    }).join('');

  }

}


function renderStats() {

  const done = tasks.filter(t => t.done).length;

  document.getElementById('stat-total').textContent = tasks.length;

  document.getElementById('stat-done').textContent  = done;

  document.getElementById('stat-undos').textContent = undoCount;

  document.getElementById('stat-redos').textContent = redoCount;

  document.getElementById('badge-all').textContent    = tasks.length;

  document.getElementById('badge-active').textContent = tasks.filter(t => !t.done).length;

  document.getElementById('badge-done').textContent   = done;

  document.getElementById('badge-high').textContent   = tasks.filter(t => t.priority==='high').length;

  document.getElementById('badge-med').textContent    = tasks.filter(t => t.priority==='med').length;

  document.getElementById('badge-low').textContent    = tasks.filter(t => t.priority==='low').length;

}


const logEntries = [];

function logAction(type, msg) {

  const now = new Date();

  const time = `${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}`;

  logEntries.unshift({ type, msg, time });

  if (logEntries.length > 20) logEntries.pop();

  document.getElementById('logEl').innerHTML = logEntries.map(e =>

    `<div class="log-entry">

      <div class="log-time">${e.time}</div>

      <div class="log-action"><span class="action-type at-${e.type}">${e.type}</span>${esc(e.msg)}</div>

    </div>`

  ).join('');

}


let toastTimer;

function showToast(msg) {

  const el = document.getElementById('toast');

  el.textContent = msg;

  el.classList.add('show');

  clearTimeout(toastTimer);

  toastTimer = setTimeout(() => el.classList.remove('show'), 1800);

}


document.addEventListener('keydown', e => {

  const tag = document.activeElement.tagName;

  if (tag === 'INPUT' || tag === 'TEXTAREA') return;

  if ((e.ctrlKey || e.metaKey) && e.key === 'z' && !e.shiftKey) { e.preventDefault(); undo(); }

  if ((e.ctrlKey || e.metaKey) && (e.key === 'y' || (e.key === 'z' && e.shiftKey))) { e.preventDefault(); redo(); }

});


renderAll();
