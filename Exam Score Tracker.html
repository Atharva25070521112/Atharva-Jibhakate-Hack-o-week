<script>

  // ─── Data structures ───────────────────────────────────────────

  // Array-based frequency array (indices 0-100)

  const freq = new Array(101).fill(0);

  // Actual score log with names

  const scoreLog = [];


  // ─── Core logic (linear traversal) ──────────────────────────────

  function computeStats() {

    let total = 0, minVal = Infinity, maxVal = -Infinity;

    let modeVal = -1, modeCount = 0, sum = 0;


    // Single linear traversal of freq array

    for (let i = 0; i <= 100; i++) {

      if (freq[i] > 0) {

        total += freq[i];

        sum += freq[i] * i;

        if (i < minVal) minVal = i;

        if (i > maxVal) maxVal = i;

        if (freq[i] > modeCount) {

          modeCount = freq[i];

          modeVal = i;

        }

      }

    }


    return {

      total,

      min: total ? minVal : null,

      max: total ? maxVal : null,

      mode: total ? modeVal : null,

      modeCount,

      avg: total ? (sum / total).toFixed(1) : null

    };

  }


  // ─── UI Update ──────────────────────────────────────────────────

  function render() {

    const stats = computeStats();


    // Stats

    document.getElementById('stat-total').textContent = stats.total || '0';

    document.getElementById('stat-min').textContent   = stats.min  ?? '—';

    document.getElementById('stat-max').textContent   = stats.max  ?? '—';

    document.getElementById('stat-mode').textContent  = stats.mode ?? '—';

    document.getElementById('stat-avg').textContent   = stats.avg  ?? '—';


    // Frequency chart

    const freqEl = document.getElementById('freq-chart');

    const occupied = [];

    for (let i = 0; i <= 100; i++) if (freq[i] > 0) occupied.push(i);


    if (!occupied.length) {

      freqEl.innerHTML = `<div class="empty"><div class="empty-icon">📊</div><p>Add scores to see<br>frequency distribution</p></div>`;

    } else {

      const maxFreq = Math.max(...occupied.map(s => freq[s]));

      freqEl.innerHTML = occupied.map(s => {

        const pct = Math.round((freq[s] / maxFreq) * 100);

        const isMode = s === stats.mode;

        return `

          <div class="freq-row">

            <div class="freq-score">${s}</div>

            <div class="freq-bar-wrap">

              <div class="freq-bar ${isMode ? 'is-mode' : ''}" style="width:${pct}%"></div>

            </div>

            <div class="freq-count">×${freq[s]}</div>

          </div>`;

      }).join('');

    }


    // Score log

    const listEl = document.getElementById('score-list');

    if (!scoreLog.length) {

      listEl.innerHTML = `<div class="empty"><div class="empty-icon">📝</div><p>Scores will appear here<br>as you add them</p></div>`;

    } else {

      listEl.innerHTML = scoreLog.map((entry, idx) => {

        let cls = '';

        if (entry.score === stats.mode && freq[entry.score] === stats.modeCount) cls = 'is-mode';

        else if (entry.score === stats.max) cls = 'is-max';

        else if (entry.score === stats.min) cls = 'is-min';

        return `

          <div class="score-tag ${cls}">

            ${entry.name ? `<span style="color:var(--muted);font-size:0.7rem">${entry.name}:</span>` : ''}

            <span>${entry.score}</span>

            <span class="remove" onclick="removeScore(${idx})">✕</span>

          </div>`;

      }).join('');

    }


    // Distribution chart (grouped by 5s)

    const distEl = document.getElementById('dist-chart');

    const buckets = [];

    for (let b = 0; b <= 95; b += 5) {

      let count = 0;

      for (let i = b; i < b + 5 && i <= 100; i++) count += freq[i];

      buckets.push({ label: `${b}`, count });

    }

    const maxB = Math.max(...buckets.map(b => b.count), 1);

    distEl.innerHTML = buckets.map(b => {

      const h = Math.round((b.count / maxB) * 90);

      const isModeRange = stats.mode !== null && parseInt(b.label) <= stats.mode && stats.mode < parseInt(b.label) + 5;

      return `

        <div class="dist-col" title="${b.label}–${parseInt(b.label)+4}: ${b.count} score(s)">

          <div class="dist-bar ${isModeRange ? 'is-mode' : ''}" style="height:${h}px"></div>

          <div class="dist-label">${b.label}</div>

        </div>`;

    }).join('');


    // Grade breakdown

    let gA=0, gB=0, gC=0, gD=0, gF=0;

    for (let i = 0; i <= 100; i++) {

      if (i >= 90) gA += freq[i];

      else if (i >= 80) gB += freq[i];

      else if (i >= 70) gC += freq[i];

      else if (i >= 60) gD += freq[i];

      else gF += freq[i];

    }

    document.getElementById('grade-a').textContent = gA;

    document.getElementById('grade-b').textContent = gB;

    document.getElementById('grade-c').textContent = gC;

    document.getElementById('grade-d').textContent = gD;

    document.getElementById('grade-f').textContent = gF;

  }


  // ─── Actions ────────────────────────────────────────────────────

  function addScore() {

    const input = document.getElementById('score-input');

    const nameInput = document.getElementById('name-input');

    const val = parseInt(input.value);


    if (isNaN(val) || val < 0 || val > 100) {

      showToast('Please enter a valid score between 0 and 100');

      input.focus();

      return;

    }


    freq[val]++;

    scoreLog.push({ score: val, name: nameInput.value.trim() });

    input.value = '';

    nameInput.value = '';

    input.focus();

    render();

  }


  function removeScore(idx) {

    const entry = scoreLog[idx];

    freq[entry.score]--;

    scoreLog.splice(idx, 1);

    render();

  }


  function clearAll() {

    if (!scoreLog.length) return;

    freq.fill(0);

    scoreLog.length = 0;

    render();

    showToast('All scores cleared');

  }


  function toggleBulk() {

    document.getElementById('bulk-area').classList.toggle('open');

  }


  function addBulk() {

    const raw = document.getElementById('bulk-input').value;

    const nums = raw.split(/[,\s\n]+/).map(s => parseInt(s.trim())).filter(n => !isNaN(n) && n >= 0 && n <= 100);

    if (!nums.length) { showToast('No valid scores found (0–100 only)'); return; }

    nums.forEach(n => { freq[n]++; scoreLog.push({ score: n, name: '' }); });

    document.getElementById('bulk-input').value = '';

    document.getElementById('bulk-area').classList.remove('open');

    render();

    showToast(`Added ${nums.length} scores`);

  }


  function showToast(msg) {

    const t = document.getElementById('toast');

    t.textContent = msg;

    t.classList.add('show');

    setTimeout(() => t.classList.remove('show'), 2500);

  }


  // Enter key support

  document.getElementById('score-input').addEventListener('keydown', e => { if (e.key === 'Enter') addScore(); });

  document.getElementById('name-input').addEventListener('keydown', e => { if (e.key === 'Enter') addScore(); });


  // Initial render

  render();
