/* ═══════════════════════════════════════════

   DOUBLY LINKED LIST IMPLEMENTATION

═══════════════════════════════════════════ */

class Node {

  constructor(title, artist, id) {

    this.title  = title;

    this.artist = artist;

    this.id     = id;

    this.prev   = null;

    this.next   = null;

  }

}


class DoublyLinkedList {

  constructor() { this.head = null; this.tail = null; this.size = 0; }


  addFront(title, artist, id) {

    const n = new Node(title, artist, id);

    if (!this.head) { this.head = this.tail = n; }

    else { n.next = this.head; this.head.prev = n; this.head = n; }

    this.size++;

  }


  addEnd(title, artist, id) {

    const n = new Node(title, artist, id);

    if (!this.tail) { this.head = this.tail = n; }

    else { n.prev = this.tail; this.tail.next = n; this.tail = n; }

    this.size++;

  }


  deleteAt(pos) {

    if (pos < 1 || pos > this.size) return null;

    let cur = this.head;

    for (let i = 1; i < pos; i++) cur = cur.next;

    if (cur.prev) cur.prev.next = cur.next; else this.head = cur.next;

    if (cur.next) cur.next.prev = cur.prev; else this.tail = cur.prev;

    this.size--;

    return cur;

  }


  deleteById(id) {

    let cur = this.head; let pos = 1;

    while (cur) { if (cur.id === id) return this.deleteAt(pos); cur = cur.next; pos++; }

    return null;

  }


  reverse() {

    let cur = this.head, tmp = null;

    while (cur) { tmp = cur.prev; cur.prev = cur.next; cur.next = tmp; cur = cur.prev; }

    if (tmp) this.head = tmp.prev;

    [this.head, this.tail] = [this.tail, this.head];

  }


  toArray() {

    const arr = []; let cur = this.head;

    while (cur) { arr.push(cur); cur = cur.next; }

    return arr;

  }


  getById(id) {

    let cur = this.head;

    while (cur) { if (cur.id === id) return cur; cur = cur.next; }

    return null;

  }

}

function togglePlay() {

  isPlaying = !isPlaying;

  document.getElementById('btn-play').textContent = isPlaying ? '⏸ Pause' : '▶ Play';

  document.getElementById('np-disc').classList.toggle('spinning', isPlaying);

  if (isPlaying) startProgress(); else stopProgress();

}


function startProgress() {

  stopProgress();

  progressTimer = setInterval(() => {

    progressVal = Math.min(progressVal + 0.5, 100);

    document.getElementById('np-bar').style.width = progressVal + '%';

    if (progressVal >= 100) { stopProgress(); isPlaying = false;

      document.getElementById('btn-play').textContent = '▶ Play';

      document.getElementById('np-disc').classList.remove('spinning');

      progressVal = 0; }

  }, 100);

}


function stopProgress() {

  clearInterval(progressTimer); progressTimer = null;

}


function playSong(id) {

  playingId = id;

  progressVal = 0;

  document.getElementById('np-bar').style.width = '0%';

  isPlaying = true;

  document.getElementById('btn-play').textContent = '⏸ Pause';

  document.getElementById('np-disc').classList.add('spinning');

  startProgress();

  render();

}


function playNext() {

  const arr = dll.toArray();

  const idx = arr.findIndex(n => n.id === playingId);

  if (idx < arr.length - 1) playSong(arr[idx+1].id);

  else showToast('End of playlist', '');

}


function playPrev() {

  const arr = dll.toArray();

  const idx = arr.findIndex(n => n.id === playingId);

  if (idx > 0) playSong(arr[idx-1].id);

  else showToast('Start of playlist', '');

}

function addSong(where) {

  const title  = document.getElementById('inp-title').value.trim();

  const artist = document.getElementById('inp-artist').value.trim();

  if (!title) { showToast('Enter a song title', 'red'); return; }

  const id = idCtr++;

  if (where === 'front') dll.addFront(title, artist || 'Unknown Artist', id);

  else dll.addEnd(title, artist || 'Unknown Artist', id);

  if (!playingId) playingId = dll.head.id;

  document.getElementById('inp-title').value = '';

  document.getElementById('inp-artist').value = '';

  shuffled = false;

  render();

  showToast(where==='front' ? '↑ Added to front' : '↓ Added to end', '');

}


function deleteByPos() {

  const pos = parseInt(document.getElementById('del-pos').value);

  if (!pos || pos < 1 || pos > dll.size) { showToast('Invalid position', 'red'); return; }

  const removed = dll.deleteAt(pos);

  if (removed && removed.id === playingId) {

    playingId = dll.head ? dll.head.id : null;

    progressVal = 0; stopProgress(); isPlaying = false;

  }

  document.getElementById('del-pos').value = '';

  render();

  showToast('✕ Removed "' + removed.title + '"', 'red');

}


function deleteById(id) {

  const node = dll.getById(id);

  if (!node) return;

  dll.deleteById(id);

  if (id === playingId) {

    playingId = dll.head ? dll.head.id : null;

    progressVal = 0; stopProgress(); isPlaying = false;

  }

  render();

  showToast('✕ Removed "' + node.title + '"', 'red');

}


function reverseShuffle() {

  if (dll.size === 0) { showToast('No songs to shuffle', 'red'); return; }

  dll.reverse();

  shuffled = !shuffled;

  document.getElementById('shuffle-btn').classList.toggle('on', shuffled);

  document.getElementById('shuffle-text').textContent = shuffled ? 'Reverse back' : 'Reverse shuffle';

  render();

  showToast(shuffled ? '⇅ Shuffled (list reversed)' : '⇅ Restored original order', 'purple');

}

function esc(s) {

  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');

}


function render() {

  const arr = dll.toArray();

  const playingNode = arr.find(n => n.id === playingId);

  const playingPos  = arr.findIndex(n => n.id === playingId) + 1;


  /* Stats */

  document.getElementById('stat-count').textContent = arr.length;

  document.getElementById('stat-pos').textContent   = playingPos || '—';

  document.getElementById('stat-mode').textContent  = shuffled ? 'ON' : 'OFF';

  document.getElementById('main-meta').textContent  = arr.length + ' song' + (arr.length!==1?'s':'') + ' · doubly linked list';

  document.getElementById('shuffled-badge').style.display = shuffled ? 'inline-flex' : 'none';


  /* Now playing */

  if (playingNode) {

    document.getElementById('np-title').textContent  = playingNode.title;

    document.getElementById('np-artist').textContent = playingNode.artist;

    document.getElementById('btn-play').textContent  = isPlaying ? '⏸ Pause' : '▶ Play';

  } else {

    document.getElementById('np-title').textContent  = '—';

    document.getElementById('np-artist').textContent = '—';

    document.getElementById('btn-play').textContent  = '▶ Play';

    document.getElementById('np-disc').classList.remove('spinning');

  }


  /* Song list */

  const list = document.getElementById('songs-list');

  if (arr.length === 0) {

    list.innerHTML = '<div class="empty-state"><div class="empty-icon">♪</div><p>No songs yet.<br/>Add one from the sidebar.</p></div>';

    return;

  }


  let html = '';

  arr.forEach((node, i) => {

    const playing = node.id === playingId;

    const prevLbl = node.prev ? node.prev.title : 'null';

    const nextLbl = node.next ? node.next.title : 'null';

    const hasPrev = !!node.prev, hasNext = !!node.next;

    html += `

      <div class="song-row${playing?' playing':''}" onclick="playSong(${node.id})"

           style="animation-delay:${i*0.04}s">

        <div class="song-num">${playing ? '♪' : i+1}</div>

        <div class="song-body">

          <div class="song-title">${esc(node.title)}</div>

          <div class="song-artist">${esc(node.artist)}</div>

          <div class="song-ptrs">

            <span class="ptr-tag${hasPrev?' has-link':''}"><span class="arrow">←</span>${esc(prevLbl)}</span>

            <span class="ptr-tag${hasNext?' has-link':''}"><span class="arrow">→</span>${esc(nextLbl)}</span>

          </div>

        </div>

        <div class="song-actions">

          <button class="act-btn remove" onclick="event.stopPropagation();deleteById(${node.id})" title="Remove">✕</button>

        </div>

      </div>`;

    if (i < arr.length - 1) {

      html += `<div class="dll-connector"><div class="dll-line"></div><span>⇅</span><div class="dll-line"></div></div>`;

    }

  });

  list.innerHTML = html;

}

let toastTimer;

function showToast(msg, type='') {

  const t = document.getElementById('toast');

  const m = document.getElementById('toast-msg');

  t.className = 'toast' + (type ? ' '+type : '');

  m.textContent = msg;

  t.classList.add('show');

  clearTimeout(toastTimer);

  toastTimer = setTimeout(() => t.classList.remove('show'), 2200);

}
